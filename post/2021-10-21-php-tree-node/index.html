二叉树的层序遍历

2021年10月21日
| leetcode

leetcode第102题 给你一个二叉树，请你返回按照层序遍历得到的节点值。（即逐层地，从左到右访问所有节点） 示例： 二叉树：[3,9,20,null,null,15,7],

  3
 / \
9   20
   /  \
  15   7

返回其层序遍历结果：

[ [3], [9,20], [15,7] ]

<?php

/**
 * Definition for a binary tree node.
 * class TreeNode {
 *     public $val = null;
 *     public $left = null;
 *     public $right = null;
 *     function __construct($val = 0, $left = null, $right = null) {
 *         $this->val = $val;
 *         $this->left = $left;
 *         $this->right = $right;
 *     }
 * }
 */
class Solution
{

    /**
     * @param TreeNode $root
     * @return Integer[][]
     */
    function levelOrder($root)
    {
        //定义返回值
        $res = [];
        //边界值的处理
        if (!$root) {
            return $res;
        }
        //定义一个队列
        $queue = [];
        //将根节点放入队列
        array_push($queue, $root);
        //定义初始层级，根节点是第一层，在返回值的数组中下标从0开始
        $level = 0;
        while (!empty($queue)) {
            foreach ($queue as $v) {
                if ($v->left) {
                    array_push($queue, $v->left);
                }
                if ($v->right) {
                    array_push($queue, $v->right);
                }
                $res[$level][] = $v->val;
                //该节点处理完毕，从队列中去除
                array_shift($queue);
            }
            //该层处理完毕，层数自增
            $level++;
        }
        return $res;
    }
}
